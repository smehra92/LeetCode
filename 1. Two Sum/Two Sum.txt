class Solution 
{
    public int[] twoSum(int[] nums, int target) 
    {
    //int[] arr;
    for(int j=0; j<nums.length; j++) // 1st for loop is for taking out a number from an array.
        {   
           for(int i=j+1; i<nums.length; i++) // 2nd loop is for taking out another number from the same array.
           {
           if(nums[j] + nums[i] == target) // Adding both the numbers and comparing the addition of them with target value.
           {
               //System.out.println(j);
               //System.out.println(i);
               //int[] arr;
               //arr[0] = i;
               //arr[1] = j;
                 return new int[] { i, j };  
           }

        }
   
    }
    
    return new int[] {};
    }
}
