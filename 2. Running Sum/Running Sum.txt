class Solution {
    public int[] runningSum(int[] nums) {
        int[] n = new int[nums.length]; //we're declaring an array and also assigning the size of the array because earlier it was giving error.
        n[0] = nums[0]; // we want first value to be same in both the arrays.
        for(int i=1; i<nums.length; i++)// so the iteration should start from index 1.
        {
            //int[] n;
            n[i] = nums[i] + n[i-1];
            
        }
        return n;
    }
}
